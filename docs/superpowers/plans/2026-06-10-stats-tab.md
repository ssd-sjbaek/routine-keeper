# 통계 탭 Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** 루틴 지킴이 앱 메인 카드에 "오늘 / 📊 통계" 탭을 추가하고, 통계 탭에서 스트릭·월간 달성 달력·루틴별 달성률과 기록 초기화 기능을 제공한다.

**Architecture:** `index.html` 단일 파일만 수정. 새 Firestore 컬렉션 없음 — 기존 `/users/{uid}/routines/{dateKey}` 문서들을 클라이언트에서 집계. 메인 카드에 탭 바를 추가하고 오늘 패널 / 통계 패널을 show/hide로 전환.

**Tech Stack:** Firebase compat SDK 10.7.1 (기존), Firestore, vanilla JS, 단일 HTML 파일

      const [y, m, d] = dateKey.split('-').map(Number);
      const dt = new Date(y, m - 1, d + 1);
      return `${dt.getFullYear()}-${String(dt.getMonth()+1).padStart(2,'0')}-${String(dt.getDate()).padStart(2,'0')}`;

    async function loadMonthRoutineChecks(year, month) {
      const m = String(month).padStart(2, '0');
      const daysInMonth = new Date(year, month, 0).getDate();
      const dateKeys = [];
      for (let d = 1; d <= daysInMonth; d++) {
        dateKeys.push(`${year}-${m}-${String(d).padStart(2, '0')}`);
      }
      const docs = await Promise.all(
        dateKeys.map(k =>
          db.collection('users').doc(currentUser.uid)
            .collection('routines').doc(k).get()
        )
      );
      const result = {};
      dateKeys.forEach((k, i) => {
        result[k] = docs[i].exists ? docs[i].data() : {};
      });
      return result;
    }

    async function loadAllRoutineChecks() {
      const snap = await db.collection('users').doc(currentUser.uid)
                            .collection('routines').get();
      const result = {};
      snap.forEach(doc => { result[doc.id] = doc.data(); });
      return result;
    }

- [ ] **Step 2: 스트릭 계산 함수 추가**

`loadAllRoutineChecks` 함수 바로 아래에 추가한다:

```javascript
    function calcStreakInfo(allChecks, routines, year, month) {
      if (routines.length === 0) return { current: 0, monthBest: 0, allTimeBest: 0 };
      const today = getTodayKey();

      const isAllDone = (key) => {
        const checks = allChecks[key] || {};
        return routines.every(r => checks[r]);
      };

      // 오늘부터 역순으로 연속 달성일 계산
      let current = 0;
      let key = today;
      while (isAllDone(key)) {
        current++;
        key = subtractOneDay(key);
        if (key < '2020-01-01') break; // 무한루프 방지
      }

      // 날짜 배열에서 가장 긴 연속 구간 계산
      const longestInKeys = (keys) => {
        let best = 0, cur = 0, prev = null;
        for (const k of keys) {
          if (isAllDone(k)) {
            cur = (prev && addOneDay(prev) === k) ? cur + 1 : 1;
            best = Math.max(best, cur);
          } else {
            cur = 0;
          }
          prev = k;
        }
        return best;
      };

      const allDates = Object.keys(allChecks).filter(k => k <= today).sort();
      const allTimeBest = longestInKeys(allDates);

      const m = String(month).padStart(2, '0');
      const monthDates = allDates.filter(k => k.startsWith(`${year}-${m}`));
      const monthBest = longestInKeys(monthDates);

      return { current, monthBest, allTimeBest };
    }
```

    function calcMonthlyRates(monthChecks, routines, year, month) {
      if (routines.length === 0) return [];
      const today = getTodayKey();
      const m = String(month).padStart(2, '0');
      const daysInMonth = new Date(year, month, 0).getDate();
      const doneCounts = Object.fromEntries(routines.map(r => [r, 0]));
      let totalDays = 0;

      for (let d = 1; d <= daysInMonth; d++) {
        const k = `${year}-${m}-${String(d).padStart(2, '0')}`;
        if (k > today) break;
        totalDays++;
        const checks = monthChecks[k] || {};
        routines.forEach(r => { if (checks[r]) doneCounts[r]++; });
      }

      return routines.map(name => ({
        name,
        rate: totalDays > 0 ? Math.round(doneCounts[name] / totalDays * 100) : 0,
      }));
    }
```

    async function resetMonthRoutines() {
      const label = `${statsYear}년 ${statsMonth}월`;
      if (!window.confirm(`${label} 루틴 기록을 모두 삭제할까요?`)) return;
      const m = String(statsMonth).padStart(2, '0');
      const daysInMonth = new Date(statsYear, statsMonth, 0).getDate();
      const batch = db.batch();
      for (let d = 1; d <= daysInMonth; d++) {
        const key = `${statsYear}-${m}-${String(d).padStart(2, '0')}`;
        batch.delete(
          db.collection('users').doc(currentUser.uid).collection('routines').doc(key)
        );
      }
      await batch.commit();
      await renderStats();
    }

    async function resetAllRoutines() {
      if (!window.confirm('모든 루틴 기록을 삭제할까요? 되돌릴 수 없어요.')) return;
      if (!window.confirm('정말요? 전체 기록이 삭제됩니다.')) return;
      const snap = await db.collection('users').doc(currentUser.uid)
                            .collection('routines').get();
      const batch = db.batch();
      snap.forEach(doc => batch.delete(doc.ref));
      await batch.commit();
      await renderStats();
    }
```

    async function renderStats() {
      const panel = document.getElementById('panel-stats');
      if (!panel) return;
      panel.innerHTML = '<div style="text-align:center;padding:20px;color:#aaa;font-size:13px">로딩 중...</div>';

      const todayDate   = new Date();
      const currentYear = todayDate.getFullYear();
      const currentMon  = todayDate.getMonth() + 1;
      const isCurrentMonth = statsYear === currentYear && statsMonth === currentMon;

      const m = String(statsMonth).padStart(2, '0');
      const daysInMonth = new Date(statsYear, statsMonth, 0).getDate();
      const today = getTodayKey();

      const [monthChecks, allChecks] = await Promise.all([
        loadMonthRoutineChecks(statsYear, statsMonth),
        loadAllRoutineChecks(),
      ]);

      const { current, monthBest, allTimeBest } =
        calcStreakInfo(allChecks, currentRoutines, statsYear, statsMonth);
      const rates = calcMonthlyRates(monthChecks, currentRoutines, statsYear, statsMonth);

      // 달력 셀 HTML
      const firstWeekday = new Date(statsYear, statsMonth - 1, 1).getDay();
      let calCells = ['일','월','화','수','목','금','토']
        .map(w => `<div class="stats-cal-header">${w}</div>`).join('');
      for (let i = 0; i < firstWeekday; i++) calCells += '<div></div>';
      for (let d = 1; d <= daysInMonth; d++) {
        const key  = `${statsYear}-${m}-${String(d).padStart(2,'0')}`;
        const done = currentRoutines.length > 0 &&
                     currentRoutines.every(r => (monthChecks[key] || {})[r]);
        let cls;
        if (key > today)      cls = 'future';
        else if (key === today) cls = done ? 'today done' : 'today';
        else                  cls = done ? 'done' : 'missed';
        calCells += `<div class="stats-cal-day ${cls}">${d}</div>`;

        <div class="rate-row">
          <div class="rate-header">
            <span>${name}</span>

          </div>
          <div class="rate-bar-bg">
            <div class="rate-bar-fill ${rate >= 70 ? 'good' : 'low'}" style="width:${rate}%"></div>
          </div>
        </div>
      `).join('');

      panel.innerHTML = `

          <div style="display:flex;align-items:center;gap:12px">
            <button class="section-btn" onclick="statsNavMonth(-1)">‹</button>
            <span style="font-size:15px;font-weight:700">${statsYear}년 ${statsMonth}월</span>

        </div>

          <div class="streak-item">
            <div class="streak-num">🔥 ${current}</div>
            <div class="streak-label">일 연속 달성 중</div>
          </div>
          <div class="streak-divider"></div>
          <div class="streak-item">
            <div class="streak-num">${monthBest}</div>
            <div class="streak-label">이번 달 최장</div>
          </div>
          <div class="streak-divider"></div>
          <div class="streak-item">
            <div class="streak-num">${allTimeBest}</div>
            <div class="streak-label">역대 최장</div>
          </div>
        </div>

        <div class="stats-section-title">📅 달성 현황</div>
        <div class="stats-cal-grid">${calCells}</div>
        <div class="stats-cal-legend">
